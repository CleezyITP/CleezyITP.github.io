Snowperson Exercise

//base snowball

hull() {

cylinder(h=20,r=40);

translate([0,0,40]) sphere(40);

}

//center snowball

translate([0,0,75]) {

sphere(r = 30, center=false);

};

//upper snowball

translate([0,0,120]) {

sphere(r = 20, center=false);

};

//carrot nose

translate([0,0,120]) rotate(a=90, v=[1,0,0]){

cylinder(h=35, r1=6, r2=1);

};

module arm(){

translate([0,0,70]) rotate(a = 70, v = [0, 1, 0]){

cube([2,3,80]);}

}

arm();

mirror([1,0,0])arm();
